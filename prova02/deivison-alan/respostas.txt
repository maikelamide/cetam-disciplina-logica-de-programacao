   notam:real

   aluno1<-"deveison"
   aluno2<-"alan"

   escreval("entre com a nota:")
   leia(vet[3])

      escreval(aluno1," esta aprovado" )

      escreval("entre com a nota de recuperação do aluno :",aluno1)
      leia (vet[1])

         escreval(aluno1," esta aprovado na recuperação")

         escreval(aluno1," vai para a recuperação final")

      fimse
